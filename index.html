function renderFilters(tab) {

  const bar = document.getElementById('filterbar');

  const configs = FILTER_CONFIGS[tab] || [];

  if(!configs.length){bar.innerHTML='';return;}

  bar.innerHTML = configs.map(f => {

    const on = state.filters[f.key] === f.val;

    return `<button class="filter-chip ${on?'on':''}" onclick="toggleFilter('${f.key}','${f.val}')">${f.label}</button>`;

  }).join('');

}

function toggleFilter(key, val) {

  state.filters[key] = state.filters[key] === val ? '' : val;

  saveState();

  renderCurrentPage();

  renderFilters(state.activeTab);

}

function applyFilters(recipes) {

  return recipes.filter(r => {

    const f = state.filters;

    if(f.cuisine && r.cuisine !== f.cuisine) return false;

    if(f.effort && r.effort !== f.effort) return false;

    if(f.marinade === 'no' && r.marinade) return false;

    if(f.time === 'quick') { const mins = parseInt(r.time); if(isNaN(mins) || mins > 15) return false; }

    if(f.time === 'medium') { const mins = parseInt(r.time); if(isNaN(mins) || mins > 30) return false; }

    if(f.cal === 'low' && r.cal >= 200) return false;

    if(f.cal === 'mid' && (r.cal < 200 || r.cal > 400)) return false;

    if(f.cal === 'vlow' && r.cal >= 100) return false;

    if(state.search) {

      const q = state.search.toLowerCase();

      if(!r.name.toLowerCase().includes(q) && !r.why.toLowerCase().includes(q) && !r.cuisine.toLowerCase().includes(q)) return false;

    }

    return true;

  });

}

// ── PAGE RENDERS ───────────────────────────────────────────────────────────────

function renderPage(containerId, recipes) {

  const el = document.getElementById(containerId);

  const filtered = applyFilters(recipes);

  if(!filtered.length) {

    el.innerHTML = `<div class="empty"><div class="empty-icon">🔍</div><p>No recipes match your filters.</p></div>`;

    return;

  }

  el.innerHTML = filtered.map(renderCard).join('');

}

function renderCurrentPage() {

  const t = state.activeTab;

  if(t==='picks') renderPage('page-picks', ALL_RECIPES.filter(r=>r.tab==='picks'));

  else if(t==='favs') {

    const favs = ALL_RECIPES.filter(r=>state.favorites[r.id]);

    const el = document.getElementById('page-favs');

    if(!favs.length) el.innerHTML=`<div class="empty"><div class="empty-icon">❤️</div><p>Tap 🤍 on any recipe to save it here.</p></div>`;

    else el.innerHTML = favs.map(renderCard).join('');

  }

  else if(t==='breakfast') renderPage('page-breakfast', ALL_RECIPES.filter(r=>r.tab==='breakfast'));

  else if(t==='desserts') renderPage('page-desserts', ALL_RECIPES.filter(r=>r.tab==='desserts'));

  else if(t==='snacks') renderPage('page-snacks', ALL_RECIPES.filter(r=>r.tab==='snacks'));

  else if(t==='all') renderPage('page-all', ALL_RECIPES);

  else if(t==='grocery') renderGrocery();

  else if(t==='ref') renderRef();

}

function render() {

  renderCurrentPage();

  renderFilters(state.activeTab);

  const favCount = Object.values(state.favorites).filter(Boolean).length;

  document.getElementById('header-sub').textContent = `${ALL_RECIPES.length} recipes · ${favCount} favorites`;

}

// ── COOK MODE ──────────────────────────────────────────────────────────────────

let cookRecipe = null, cookIdx = 0, timerSecs = 0, timerRunning = false, timerInterval = null;

function openCook(id) {

  cookRecipe = ALL_RECIPES.find(r=>r.id===id);

  cookIdx = 0;

  document.getElementById('cook-title').textContent = cookRecipe.name;

  document.getElementById('cookmode').classList.add('open');

  updateCookDisplay();

}

function closeCook() {

  document.getElementById('cookmode').classList.remove('open');

  clearInterval(timerInterval);

  timerRunning = false;

  document.querySelector('#cook-timer-btns button:last-child').textContent = '▶ Start';

}

function updateCookDisplay() {

  const steps = cookRecipe.steps;

  document.getElementById('cook-step-num').textContent = `Step ${cookIdx+1} of ${steps.length}`;

  document.getElementById('cook-step-text').textContent = steps[cookIdx];

  document.getElementById('cook-prev').style.opacity = cookIdx===0?'0.3':'1';

  document.getElementById('cook-next').textContent = cookIdx===steps.length-1 ? '✓ Done' : 'Next →';

}

function cookStep(dir) {

  const steps = cookRecipe.steps;

  if(dir===1 && cookIdx===steps.length-1){closeCook();return;}

  cookIdx = Math.max(0, Math.min(steps.length-1, cookIdx+dir));

  updateCookDisplay();

}

function setTimer(mins) { timerSecs = mins*60; updateTimerDisplay(); }

function toggleTimer() {

  const btn = document.querySelector('#cook-timer-btns button:last-child');

  if(timerRunning) { clearInterval(timerInterval); timerRunning=false; btn.textContent='▶ Start'; }

  else if(timerSecs>0) {

    timerRunning=true; btn.textContent='⏸ Pause';

    timerInterval = setInterval(()=>{ if(timerSecs>0){timerSecs--;updateTimerDisplay();}else{clearInterval(timerInterval);timerRunning=false;btn.textContent='▶ Start';document.getElementById('cook-timer-display').textContent='DONE!';} },1000);

  }

}

function updateTimerDisplay() {

  const m = Math.floor(timerSecs/60).toString().padStart(2,'0');

  const s = (timerSecs%60).toString().padStart(2,'0');

  document.getElementById('cook-timer-display').textContent = `${m}:${s}`;

}
